function checkOddEven(number) {
  if (number % 2 === 0) {
    console.log(number + " is even.");
  } else {
    console.log(number + " is odd.");
  }
}

checkOddEven(5);
checkOddEven(10);
checkOddEven(7);
checkOddEven(2);
