-- Write your query below
SELECT
    c.country_id,
    c.country,
    COUNT(r.rental_id) AS rental_count
FROM country c
JOIN city ci
    ON c.country_id = ci.country_id
JOIN address a
    ON ci.city_id = a.city_id
JOIN customer cu
    ON a.address_id = cu.address_id
JOIN rental r
    ON cu.customer_id = r.customer_id
GROUP BY
    c.country_id,
    c.country
ORDER BY
    rental_count DESC
LIMIT 3;
